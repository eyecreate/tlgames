function removeMemberFromPlayersParty(playerId, removingMemberId)
  player = Player(playerId)
  local party = player:getParty()

  for k,v in pairs(party:getMembers()) do
    if v.getGuid() == Player(removingMemberId).getGuid() then
      party:removeMember(Player(removingMemberId))
    end
  end
end
